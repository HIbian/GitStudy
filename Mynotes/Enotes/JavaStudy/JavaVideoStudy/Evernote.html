创建时间：2018/8/1 19:12
作者：41671127@qq.com

  • 继承
    • 从已有的类创建一个新的类的过程
      • 面向对象三大特征之一
      • 被继承的类为父类，继承父类的类为子类
      • 子类可以直接调用父类的属性和方法
    • protected
      • 子类可以使用 
    • 只能有一个父类
    • 可以多层继承
    • 构造方法不能被继承
    • 创建子类对象时，父类的默认构造方法也会被调用（子类会使用父类的数据，通过父类构造方法初始化数据）
      • 实例化子类对象，会先调用父类的构造方法，父类中没有没有默认构造方法，子类需要通过Super()显式地调用父类含参构造方法，并且只能写在子类构造的第一句
    • 继承的好处
      • 通过继承可以实现代码重用，提高复用性
      • 提高代码维护性
      • 让类与类之间产生关系，多态的前提--不同环境不同形态
    • 开发原则：高内聚，低耦合
    • 子类进行实例化，先会让其父类进行初始化（调用父类构造方法），再进行子类的实例化操作
    • 子类进行实例化，执行先执行子类构造方法，子类构造会在第一句语句默认调用父类构造，若没有构造，需要通过Super调用父类构造
  • 方法重写（override methead）
    • 发生在子类中，格式必须与父类中的方法完全一致
    • 子类抛出的异常不能超过父类抛出的异常（子类抛出的异常小于父类）
    • 子类方法访问级别不能低于父类方法访问级别
    • 父类方法使用private，static，final，子类不能重写
  • super关键字
    • 调用父类属性
    • 调用父类方法
    • 调用父类构造方法，必须写在子类构造方法第一条语句，不显示的写出来，默认调用父类无参构造方法
  • final关键字
    • 声明一个常量
      • 修饰属性或者局部变量
      • 常量使用全大写命名 eg NUM
      • 常量必须在定义时或者在默认构造器中初始化
package com.test.java;

    • 声明一个方法
      • 最终方法，能被子类继承，但不能被改写
    • 声明一个类
      • 最终类，不能被继承，没有子类final修饰的类无法被继承
    • 方法的参数使用final，方法的内部不能改变方法的值
  • 抽象类
    • 具有很多相同特征的类可以抽象为抽象类
    • 抽象方法：只有申明，没有实现
      • 子类必须实现（重写）所有的方法
      • 抽象类可以没有抽象方法，有抽象方法的类必须是抽象类
      • 抽象类可以继承抽象类，可以不实现父类的抽象方法
      • 抽象类可以有方法和属性
      • 抽象类不能被实例化
      • 抽象类不能申明为final
      • 抽象类可以有构造方法
  • 接口
    • 定义格式
      • 使用关键字interface
interface 接口名称{
    全局常量；
    抽象方法
}
    • 和类平级
      • 接口中可以声明常量和抽象方法，JDK1.8以后可以定义默认的实现方法
      • 接口可以继承多个接口 extends I1，I2，I3.....
      • 具体的类实现接口使用implements关键字
      • 一个类可以实现多个接口
      • 抽象类可以不实现接口的方法
      • 具体类必须实现接口的所有方法
      • 接口不能有构造方法，接口不能被实例化
  • 设计原则
    • 对修改关闭，对扩展开放
    • 面向接口编程
  • instanceof
    • 用于检查是否为指定类型的对象
    • 注意 instanceof 指定类和指定类的父类都能通过

class NodeManager{
     
     private Node root;
     private int currctIndex =0;
     
     //添加
     public void add(int data) {
          if (root == null) {
               root = new Node(data);
          }
          root.addNode(data);
     }
     //删除
     public void del(int data) {
          if (root.getData()==data) {
               root = root.next;
          }else {
               root.delNode(data);
          }
     }
     //输出所有
     public void print() {
          root.printNode();
     }
     //查找
     public boolean find(int data) {
          if (root.getData()==data) {
               return true;
          }else {
               return root.findNode(data);
          }
     }
     //修改
     public void update(int olddata,int newdata) {
          if (root.getData()==olddata) {
               root.setData(newdata);
          }
          else {
               root.updateNode(olddata, newdata);
          }
     }
     //插入
     public void insert(int index,int data) {
          if (index==0) {
               Node inserNode = new Node(data);
               //Node temp = new Node();
               //temp = inserNode;
               inserNode.next = root.next;
               root.next =inserNode;
          }
          else {
               currctIndex++;
               root.next.insertNode(index, data);
          }
          currctIndex=0;
     }
     
     private class Node{
          private int data;
          private Node next;
          
          //构造
          public Node() {}
          public Node(int data) {
               super();
               this.data = data;
          }
          //添加
          public void addNode(int data) {
               if (next==null) {
                    this.next= new Node(data);
               }else {
                    //递归
                    this.next.addNode(data);
               }
          }
          //删除
          public void delNode(int data) {
               if (this.next!=null) {
                    if (this.next.data ==data) {
                         this.next = this.next.next;
                    }else {
                         this.next.delNode(data);
                    }
               }
          }

          public void printNode() {

public V put(K key, V value) {
        //初始化Hash数组
        if (table == EMPTY_TABLE) {
            inflateTable(threshold);
        }
        if (key == null)
            return putForNullKey(value);
        
        //获取元素Hash值
        int hash = hash(key);
        //通过hash值获取到数组内的下标
        int i = indexFor(hash, table.length);
        
        for (Entry<K,V> e = table[i]; e != null; e = e.next) {
            Object k;
            if (e.hash == hash && ((k = e.key) == key || key.equals(k))) {
                V oldValue = e.value;
                e.value = value;
                e.recordAccess(this);
                return oldValue;
            }
        }

        modCount++;
        //添加映射关系
        addEntry(hash, key, value, i);
        return null;
}

void addEntry(int hash, K key, V value, int bucketIndex) {
        //判断是否扩容
        if ((size >= threshold) && (null != table[bucketIndex])) {
            resize(2 * table.length);
            hash = (null != key) ? hash(key) : 0;
            bucketIndex = indexFor(hash, table.length);
        }
        //创建映射关系
        createEntry(hash, key, value, bucketIndex);
}

        FileInputStream in = new FileInputStream(scrFile);
        FileOutputStream out = new FileOutputStream(copyedFile,true);//连续读写

        int len = -1;
        while ((len = in.read(buffer))!=-1) {

            hasCopyedSize_kb+=buffer_size/1024;
            //进度条
            Progressibility(hasCopyedSize_kb, size_kb);
        }

        out.close();
        in.close();

package IOtest;
import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
public class FileDivisionDemo {
     final static long DEFAULTSIZE = 1024*1024;
     public static void main(String[] args) throws IOException {
          File file = new File("F:\\test\\sample.zip");
          TheDivision(file,DEFAULTSIZE*3);
            merge();
     }
     /**
      * 文件分割
      *
      * @param file    要分割的文件
      * @param cutsize 分割后每个文件的大小
      * @throws IOException
      */
     public static void TheDivision(File file, long cutsize) throws IOException {
          // 判断文件是否存在
          if (file == null) {
               return;
          }
          // 计算分割的个数
          int num = (file.length() % cutsize == 0) ? (int) (file.length() / cutsize)
                    : (int) (file.length() / cutsize + 1);
          // 文件输入流
          BufferedInputStream in = new BufferedInputStream(new FileInputStream(file));
          //文件输出流
          BufferedOutputStream out = null;
          //每次读取长度
          byte[] bytes = null;
          //每次读取实际长度
          int len = -1;
          //每次分割需要读取的次数
          int count = 0;
          
          for (int i = 0; i < num; i++) {
               out = new BufferedOutputStream(new FileOutputStream(new File("F:\\test\\"+(i+1)+"---"+file.getName())));
               if (cutsize<DEFAULTSIZE) {//每次分割大小小于1024
                    bytes = new byte[(int)cutsize];
                    count =1;
               }else {
                    bytes = new byte[(int)DEFAULTSIZE];
                    count = (int)cutsize/(int)DEFAULTSIZE;//分割文件需要搬运的次数
               }
               //输出分割文件
               while(count>0&&((len=in.read(bytes))!=-1)) {
                    out.write(bytes,0,len);
                    out.flush();//
                    count--;
               }
               //输出剩余的分割文件
               if (cutsize%DEFAULTSIZE!=0) {
                    //不能读取多了,数组大小刚好为剩余部分的大小
                    bytes = new byte[(int)cutsize%(int)DEFAULTSIZE];
                    in.read(bytes);
                    out.write(bytes,0,bytes.length);
                    out.flush();
               }
               out.close();
          }
          in.close();    
     }
     public static void merge() throws IOException {
          InputStream in1 = new FileInputStream(new File("F:\\test\\1---sample.zip"));
          InputStream in2 = new FileInputStream(new File("F:\\test\\2---sample.zip"));
          InputStream in3 = new FileInputStream(new File("F:\\test\\3---sample.zip"));
          InputStream in4 = new FileInputStream(new File("F:\\test\\4---sample.zip"));
          InputStream in5 = new FileInputStream(new File("F:\\test\\5---sample.zip"));
          Vector<InputStream> v = new Vector<>();
          v.add(in1);
          v.add(in2);
          v.add(in3);
          v.add(in4);
          v.add(in5);
          Enumeration<InputStream> es = v.elements();
          //合并流
          SequenceInputStream sis = new SequenceInputStream(es);
          BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(new File("F:\\sample.zip")));
          byte[] bs = new byte[1024];
          int len = -1;
          while ((len=sis.read(bs))!=-1) {
               bos.write(bs, 0, len);
          }
          
          bos.close();
          sis.close();
          System.out.println("done");
          
     }
}

package 网络编程;
import java.io.BufferedOutputStream;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.PrintStream;
importjava.net.ServerSocket;
importjava.net.Socket;
public class ECHOServerDemo {
         public static void main(String[] args) throws IOException {
                 //创建服务器并且指定端口号
                 ServerSocket server = new ServerSocket(6666);
                 System.out.println("服务器已经启动，等待连接");
                 //等待客户端连接，线程阻塞
                 Socket socket = server.accept();
                 System.out.println("连接成功:"+server.getInetAddress());
                 //读取客户端数据
                 BufferedReader br = new BufferedReader(new InputStreamReader(socket.getInputStream()));
                 String info = br.readLine();
                 System.out.println("服务器端info："+info);
                 
                 //返回数据
                 PrintStream ps = new PrintStream(new BufferedOutputStream(socket.getOutputStream()));
                 ps.println("echo"+info);
                 ps.flush();
                 ps.close();
                 br.close();
                 server.close();
         }
}
