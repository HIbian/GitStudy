     public static void Bubble_Plus() {
          int[] nums = new int[20];
          Random random =new Random();
          //临时变量
          int temp = 0;
          //初始化一个随机数组
          for (int i = 0; i < nums.length; i++) {
               nums[i] = random.nextInt(100)+1;
          }
          //打印数组
          for (int i : nums) {
               System.out.print(i+" ");
          }
          System.out.print("\n");
          //排序(从小到大)
          int left =0;
          int right = nums.length-1;
          while (left<=right) {
               //从左往右
               for (int i = left; i < right; i++) {
                    if (nums[i]>nums[i+1]) {
                         temp =nums[i+1];
                         nums[i+1] = nums[i];
                         nums[i] = temp;
                    }
               }
               left++;
               //从右往左
               for (int i = right; i >= left; i--) {
                    if (nums[i]<nums[i-1]) {
                         temp =nums[i-1];
                         nums[i-1] = nums[i];
                         nums[i] = temp;
                    }
               }
               right--;
          }
          //打印数组
          for (int i : nums) {
               System.out.print(i+" ");
          }
     }
